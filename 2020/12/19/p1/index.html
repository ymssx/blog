# 传感器驱动代码的设计模式思考

在搜寻网络上的传感器驱动代码时，能看到许多面向过程式的代码。同时我在围观机械专业的同学时，他们的代码也多是面向过程式的，整体看起来十分混乱也不方便维护。 作为一名软件工程师，当然要对自己的代码质量严格要求。

首先我们如何去定义一个传感器呢，对软件开发来说，传感器的硬件逻辑其实是我们不需要关心的，也就是说对于一个传感器我们只需要关心它能提供的功能就行了。

我首先想到的是用函数式编程直接暴露核心api，比如对于温度传感器，我只需要export一个getTemperature函数来获取温度即可，其电路逻辑代码全部在函数内部自行维护，再比如一个RGB的LED灯，我需要的是一个setEGB函数来设置LED灯的颜色即可，其中如何设置PWM，如何在程序结束自动cleanup由函数内部自行维护。

理想的使用方式

import { getTemperature, setRGB } from 'sensor';

const temperature = getTemperature();
if (temperature > 30) {
  // 当温度过高时，设置led发出红光，否则绿光
  setRGB('red');
} else {
  setRGB('green');
}

考虑到经常需要连接多个同样的传感器，使用api时还需要传入针脚等配置信息，所以最终还是选择面向对象式编程，即一个传感器是一个对象。

import { Temperature, RGB_LEG } from 'sensor';

const temperature = new Temperature({ channel = [18, 15, 14], mode = 'BCM' });
const led = new RGB_LED({ channel = [18, 15, 14], mode = 'BCM' });

if (temperature.geTemperature() > 30) {
  // 当温度过高时，设置led发出红光，否则绿光
  led.setRGB('red');
} else {
  led.setRGB('green');
}

这种编程模式简单直观，我们只需要专注于业务逻辑即可

#!/usr/bin/env python
# encoding: utf-8
# 超声波测距
 
import RPi.GPIO
from atexit import register
import time

class Wave:
  # channel -- [trig, echo]
  def __init__(self, channel = [23, 24], mode = RPi.GPIO.BCM):
    self.channel = channel

    trig_channel, echo_channel = channel

    RPi.GPIO.setmode(mode)
    RPi.GPIO.setwarnings(False)
    
    RPi.GPIO.setup(trig_channel, RPi.GPIO.OUT)
    RPi.GPIO.setup(echo_channel, RPi.GPIO.IN)

    self.control = RPi.GPIO

    ## 程序执行结束前自动清理
    register(self.stop)

  def distance(self):
    try:
      control = self.control
      trig_channel, echo_channel = self.channel

      control.output(trig_channel, 0)
      time.sleep(0.000002)

      control.output(trig_channel, 1)
      time.sleep(0.00001)
      control.output(trig_channel, 0)

      while control.input(echo_channel) == 0:
        pass
      start = time.time()

      while control.input(echo_channel) == 1:
        pass
      now = time.time()
      during = now - start

      WAVE_SPEED = 340

      return during * WAVE_SPEED / 2 * 100

    except KeyboardInterrupt:
      self.stop()

  def on_distance_change(self, callback, sensi = 1, space = 0.5):
    last_dis = 0
    while True:
      dis = self.distance()
      if abs(dis - last_dis) > sensi:
        callback(dis)
      last_dis = dis

      time.sleep(space)

  def stop(self):
    RPi.GPIO.cleanup()
    print('cleanup!')


if __name__ == "__main__":
  wave = Wave()

  def callback(dis):
    print(dis)

  wave.on_distance_change(callback)

#!/usr/bin/env python
# encoding: utf-8
# 控制RGB灯发出指定rgb颜色
 
import RPi.GPIO
from atexit import register
import time

class RGB_LED:
  # channel -- [R, G, B]
  def __init__(self, channel = [18, 15, 14], mode = RPi.GPIO.BCM):
    self.channel = channel
    R_channel, G_channel, B_channel = channel

    RPi.GPIO.setmode(mode)
    RPi.GPIO.setwarnings(False)
    
    RPi.GPIO.setup(R_channel, RPi.GPIO.OUT)
    RPi.GPIO.setup(G_channel, RPi.GPIO.OUT)
    RPi.GPIO.setup(B_channel, RPi.GPIO.OUT)

    RPi.GPIO.output(R_channel, RPi.GPIO.LOW)
    RPi.GPIO.output(G_channel, RPi.GPIO.LOW)
    RPi.GPIO.output(B_channel, RPi.GPIO.LOW)
    
    self.pwmR = RPi.GPIO.PWM(R_channel, 2000)
    self.pwmG = RPi.GPIO.PWM(G_channel, 1999)
    self.pwmB = RPi.GPIO.PWM(B_channel, 5000)
    
    self.pwmR.start(0)
    self.pwmG.start(0)
    self.pwmB.start(0)

    register(self.stop)

  def setRGB(self, R, G, B):
    self.color = [R, G, B]

    try:
      self.pwmR.ChangeDutyCycle(R / 51 * 20)
      self.pwmG.ChangeDutyCycle(G / 51 * 20)
      self.pwmB.ChangeDutyCycle(B / 51 * 20)
    except KeyboardInterrupt:
      self.stop()

  def stop(self):
    self.pwmR.stop()
    self.pwmG.stop()
    self.pwmB.stop()
    
    RPi.GPIO.cleanup()
    print('cleanup!')


if __name__ == "__main__":
  led = RGB_LED()
  led.setRGB(255, 215, 2)
  time.sleep(3)

#!/usr/bin/env python
# encoding: utf-8
# 控制激光发射器的发射停止
 
import RPi.GPIO
from atexit import register
import time

class Beam:
  def __init__(self, channel = 18, mode = RPi.GPIO.BCM):
    self.channel = channel

    RPi.GPIO.setmode(mode)
    RPi.GPIO.setwarnings(False)
    
    RPi.GPIO.setup(channel, RPi.GPIO.OUT)
    RPi.GPIO.output(channel, RPi.GPIO.LOW)

    self.control = RPi.GPIO

    register(self.stop)

  def emit(self):
    try:
      self.control.output(self.channel, RPi.GPIO.HIGH)
    except KeyboardInterrupt:
      self.stop()

  def stop(self):
    self.control.output(self.channel, RPi.GPIO.LOW)
    RPi.GPIO.cleanup()
    print('cleanup!')


if __name__ == "__main__":
  beam = Beam()
  beam.emit()
  time.sleep(3)
